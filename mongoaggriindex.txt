db.students.drop();

// Step 1: Insert multiple student documents (Create)

  {id: 1, name: "ABC", age: 20, department: "Computer Engineering", courses: {}}, 
  {id: 2, name: "Pranav", age: 20, department: "Electrical Engineering"}
]);

// Step 2: Find students in Electrical Engineering (Read with condition)
db.students.find({department: "Electrical Engineering"});

// Step 3: Find only name and department for all students (Projection)
db.students.find({}, {name: 1, department: 1});

// Step 4: Update one student (Update)

// Step 5: Check updated student with id 1 (Read)

// Step 6: Update all students to add 'college' field (Update many)

// Step 7: Check all students after update (Read)
db.students.find();

// Step 8: Delete one student with id 1 (Delete)
db.students.deleteOne({id: 1});
