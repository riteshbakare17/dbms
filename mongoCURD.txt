db.students.find({department: "Electrical Engineering"});

db.students.find({}, {name: 1, department: 1});

db.students.updateOne({id: 1}, {$set: {age: 21, department: "AI_DS"}});

// Find the student with id = 1 to check the updated values
db.students.find({id: 1});

db.students.find();

db.students.deleteOne({id: 1});

// Delete all students whose 'college' field is "AISSMS IOIT"
